**Step 1 — Create `ninjatrader/Custom/AddOns/DEEP6/Scoring/ScorerSharedState.cs`** (NT8-API-free):
```csharp
namespace NinjaTrader.NinjaScript.AddOns.DEEP6.Scoring
{
    // Single-writer (indicator) / single-reader (strategy) latest-value latch.
    // Keyed by instrument symbol to support multi-chart scenarios.
    public static class ScorerSharedState
    {
        private static readonly System.Collections.Concurrent.ConcurrentDictionary<string, ScorerResult> _latest =
            new System.Collections.Concurrent.ConcurrentDictionary<string, ScorerResult>();
        private static readonly System.Collections.Concurrent.ConcurrentDictionary<string, int> _latestBarIdx =
            new System.Collections.Concurrent.ConcurrentDictionary<string, int>();

        public static void Publish(string symbol, int barIdx, ScorerResult result)
        {
            if (symbol == null || result == null) return;
            _latest[symbol] = result;
            _latestBarIdx[symbol] = barIdx;
        }

        public static ScorerResult Latest(string symbol)
        {
            ScorerResult r;
            return _latest.TryGetValue(symbol ?? "", out r) ? r : null;
        }

        public static int LatestBarIndex(string symbol)
        {
            int i;
            return _latestBarIdx.TryGetValue(symbol ?? "", out i) ? i : -1;
        }
    }
}
```
