    string python = Environment.GetEnvironmentVariable("PYTHON3_PATH");
    if (string.IsNullOrEmpty(python)) python = "python3";
    string repoRoot = /* resolve from assembly location — walk up until find deep6/ dir */;

    var psi = new ProcessStartInfo

        WorkingDirectory = repoRoot,

        RedirectStandardOutput = true,
        RedirectStandardError = true,
        UseShellExecute = false,

    using (var sr = new StreamReader(ndjsonPath))

    if (!proc.WaitForExit(timeoutMs))

```csharp

[TestCase("scoring-session-02")]

public void Parity_ScoringSession_WithinEnvelope(string sessionName)
{
    var path = Path.Combine("fixtures", "scoring", "sessions", sessionName + ".ndjson");
    var scoredBars = CaptureReplayLoader.LoadScoredBars(path).ToList();
    var pythonResults = RunPython(path).ToDictionary(r => r.barIdx);

    var divergences = new List<string>();
    foreach (var sbr in scoredBars)
    {
        var cs = ConfluenceScorer.Score(
            sbr.Signals, sbr.BarsSinceOpen, sbr.BarDelta, sbr.BarClose,
            zoneScore: sbr.ZoneScore, zoneDistTicks: sbr.ZoneDistTicks);
        var py = pythonResults[sbr.BarIdx];
        double delta = Math.Abs(cs.TotalScore - py.score);
        if (delta > 0.05) divergences.Add($"bar {sbr.BarIdx}: C#={cs.TotalScore:F4} vs Py={py.score:F4} Δ={delta:F4}");
        if (cs.Tier.ToString() != py.tier) divergences.Add($"bar {sbr.BarIdx}: tier mismatch C#={cs.Tier} vs Py={py.tier}");
    }
    Assert.IsEmpty(divergences, $"{sessionName}: {divergences.Count} divergences:\n  " + string.Join("\n  ", divergences));
}
```
