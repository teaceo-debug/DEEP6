private void EvaluateEntry(int barIdx, ScorerResult scored)

}

public static class ScorerEntryGate
{
    public enum GateOutcome { NoScore, NoDirection, BelowScore, BelowTier, Passed }
    public static GateOutcome Evaluate(ScorerResult scored, double scoreThreshold, SignalTier minTier)
    {
        if (scored == null) return GateOutcome.NoScore;
        if (scored.Direction == 0) return GateOutcome.NoDirection;
        if (scored.TotalScore < scoreThreshold) return GateOutcome.BelowScore;
        if ((int)scored.Tier < (int)minTier) return GateOutcome.BelowTier;

}
