The Zustand store exposes:
```typescript
interface TradingState {
  bars: RingBuffer<FootprintBar>;       // 500 capacity — MUTABLE, read via getState() for Canvas
  signals: RingBuffer<SignalEvent>;     // 200 capacity
  tape: RingBuffer<TapeEntry>;          // 50 capacity (tape row shape — placeholder for Wave 3)
  score: ScoreSlice;                    // latest LiveScoreMessage denormalized
  status: StatusSlice;                  // latest LiveStatusMessage denormalized
  lastBarVersion: number;               // bumped on each bar push — Canvas useSyncExternalStore trigger
  // actions
  pushBar: (bar: FootprintBar) => void;
  pushSignal: (sig: SignalEvent) => void;
  pushTape: (entry: TapeEntry) => void;
  setScore: (s: LiveScoreMessage) => void;
  setStatus: (s: LiveStatusMessage) => void;
  dispatch: (msg: LiveMessage) => void;   // type-discriminated router
}
```

Key architectural choice (per RESEARCH.md §Architecture Patterns): bars live in a **mutable RingBuffer instance** held directly inside the Zustand state. Readers that must avoid re-renders at bar-push frequency (the Canvas renderer) subscribe to `lastBarVersion` and read the ring buffer via `useTradingStore.getState().bars.toArray()`. React components that need reactive bar slices use `useTradingStore(s => s.bars.toArray().slice(-30))` and accept the re-render cost.

    **Step 2 — `dashboard/store/ringBuffer.ts`.** Small mutable class (FIFO overwrite):
    ```typescript
    export class RingBuffer<T> {
      private buf: (T | undefined)[];
      private head = 0;        // next write index
      private _size = 0;
      constructor(public readonly capacity: number) {
        if (capacity <= 0) throw new Error('capacity must be > 0');
        this.buf = new Array(capacity);
      }
      push(item: T): void {
        this.buf[this.head] = item;
        this.head = (this.head + 1) % this.capacity;
        if (this._size < this.capacity) this._size++;
      }
      get size(): number { return this._size; }
      get latest(): T | undefined {
        if (this._size === 0) return undefined;
        const idx = (this.head - 1 + this.capacity) % this.capacity;
        return this.buf[idx];
      }
      toArray(): T[] {
        if (this._size < this.capacity) {
          return this.buf.slice(0, this._size) as T[];
        }
        return [...this.buf.slice(this.head), ...this.buf.slice(0, this.head)] as T[];
      }
      clear(): void {
        this.buf = new Array(this.capacity);
        this.head = 0;
        this._size = 0;
      }
    }
    ```

      pushBar: (b: FootprintBar) => void;
      pushSignal: (s: SignalEvent) => void;
      pushTape: (t: TapeEntry) => void;
      setScore: (s: LiveScoreMessage) => void;
      setStatus: (s: LiveStatusMessage) => void;
      dispatch: (msg: LiveMessage) => void;
    }

    const INIT_SCORE: ScoreSlice = {
      totalScore: 0, tier: 'QUIET', direction: 0,
      categoriesFiring: [], categoryScores: {},
      kronosBias: 0, kronosDirection: 'NEUTRAL', gexRegime: 'NEUTRAL',
    };
    const INIT_STATUS: StatusSlice = {
      connected: false, pnl: 0, circuitBreakerActive: false, feedStale: false, lastTs: 0,
    };

    export const useTradingStore = create<TradingState>()(subscribeWithSelector((set, get) => ({
      bars: new RingBuffer<FootprintBar>(BAR_CAPACITY),
      signals: new RingBuffer<SignalEvent>(SIGNAL_CAPACITY),
      tape: new RingBuffer<TapeEntry>(TAPE_CAPACITY),
      score: INIT_SCORE,
      status: INIT_STATUS,
      lastBarVersion: 0,
      lastSignalVersion: 0,

      pushBar: (b) => {
        get().bars.push(b);
        set((s) => ({ lastBarVersion: s.lastBarVersion + 1 }));
      },
      pushSignal: (sig) => {
        get().signals.push(sig);
        set((s) => ({ lastSignalVersion: s.lastSignalVersion + 1 }));
      },
      pushTape: (t) => { get().tape.push(t); },
      setScore: (m) => set({
        score: {
          totalScore: m.total_score, tier: m.tier, direction: m.direction,
          categoriesFiring: m.categories_firing, categoryScores: m.category_scores,
          kronosBias: m.kronos_bias, kronosDirection: m.kronos_direction,
          gexRegime: m.gex_regime,
        },
      }),
      setStatus: (m) => set({
        status: {
          connected: m.connected, pnl: m.pnl,
          circuitBreakerActive: m.circuit_breaker_active,
          feedStale: m.feed_stale, lastTs: m.ts,
        },
      }),
      dispatch: (msg) => {
        switch (msg.type) {
          case 'bar':    get().pushBar(msg.bar); break;
          case 'signal': get().pushSignal(msg.event); break;
          case 'score':  get().setScore(msg); break;
          case 'status': get().setStatus(msg); break;
          default: {
            // eslint-disable-next-line no-console
            console.warn('[tradingStore] unknown message type', (msg as { type?: string }).type);
          }
        }
      },
    })));
