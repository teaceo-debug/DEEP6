    **Step 2 — `dashboard/hooks/useWebSocket.ts`:**

    ```typescript
    'use client';
    import { useEffect, useRef } from 'react';
    import { useTradingStore } from '@/store/tradingStore';
    import type { LiveMessage } from '@/types/deep6';

    const BACKOFF_SEQUENCE = [1000, 2000, 4000, 8000, 16000, 30000] as const;

    export function useWebSocket(url: string) {
      const reconnectAttempt = useRef(0);
      const wsRef = useRef<WebSocket | null>(null);
      const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
      const closedIntentionallyRef = useRef(false);

      useEffect(() => {
        closedIntentionallyRef.current = false;

        const connect = () => {
          if (typeof document !== 'undefined' && document.visibilityState === 'hidden') {
            // Do not connect while tab is hidden (per UI-SPEC §WebSocket Protocol Contract)
            return;
          }
          const ws = new WebSocket(url);
          wsRef.current = ws;
          ws.onopen = () => {
            reconnectAttempt.current = 0;
            const setStatus = useTradingStore.getState().setStatus;
            setStatus({ type: 'status', connected: true, pnl: 0, circuit_breaker_active: false, feed_stale: false, ts: Date.now() / 1000 });
          };
          ws.onmessage = (ev) => {
            try {
              const msg = JSON.parse(ev.data) as LiveMessage;
              if (!msg || typeof (msg as {type?: unknown}).type !== 'string') return;
              useTradingStore.getState().dispatch(msg);
            } catch {
              // drop malformed payload
            }
          };
          ws.onerror = () => { /* handled by onclose */ };
          ws.onclose = () => {
            const setStatus = useTradingStore.getState().setStatus;
            setStatus({ type: 'status', connected: false, pnl: useTradingStore.getState().status.pnl, circuit_breaker_active: useTradingStore.getState().status.circuitBreakerActive, feed_stale: false, ts: Date.now() / 1000 });
            if (closedIntentionallyRef.current) return;
            scheduleReconnect();
          };
        };

        const scheduleReconnect = () => {
          if (typeof document !== 'undefined' && document.visibilityState === 'hidden') return;
          const delay = BACKOFF_SEQUENCE[Math.min(reconnectAttempt.current, BACKOFF_SEQUENCE.length - 1)];
          reconnectAttempt.current += 1;
          timerRef.current = setTimeout(connect, delay);
        };

        const onVis = () => {
          if (document.visibilityState === 'visible'
              && (!wsRef.current || wsRef.current.readyState === WebSocket.CLOSED)) {
            reconnectAttempt.current = 0;
            connect();
          }
        };
        if (typeof document !== 'undefined') document.addEventListener('visibilitychange', onVis);

        connect();

        return () => {
          closedIntentionallyRef.current = true;
          if (timerRef.current) clearTimeout(timerRef.current);
          wsRef.current?.close();
          if (typeof document !== 'undefined') document.removeEventListener('visibilitychange', onVis);
        };
      }, [url]);
    }
    ```

    ```tsx
    'use client';
    import { useEffect, useRef } from 'react';
    import { createChart, ColorType, type IChartApi, type ISeriesApi } from 'lightweight-charts';
    import { useTradingStore } from '@/store/tradingStore';
    import type { FootprintBar } from '@/types/deep6';
    import { FootprintSeries, footprintSeriesDefaults } from '@/lib/lw-charts/FootprintSeries';
    import { ZoneOverlay } from './ZoneOverlay';

    export function FootprintChart() {
      const hostRef = useRef<HTMLDivElement | null>(null);
      const chartRef = useRef<IChartApi | null>(null);
      const seriesRef = useRef<ISeriesApi<'Custom'> | null>(null);

      useEffect(() => {
        if (!hostRef.current) return;
        const chart = createChart(hostRef.current, {
          layout: { background: { type: ColorType.Solid, color: '#0a0a0f' }, textColor: '#6b7280' },
          grid:    { horzLines: { color: '#1e1e2e' }, vertLines: { color: '#1e1e2e' } },
          crosshair: { mode: 1 /* Magnet */ },
          rightPriceScale: { borderColor: '#1e1e2e' },
          timeScale:       { borderColor: '#1e1e2e', timeVisible: true, secondsVisible: false },
          autoSize: true,
        });
        chartRef.current = chart;
        const series = chart.addCustomSeries(new FootprintSeries(), footprintSeriesDefaults);
        seriesRef.current = series;

        // Initial data
        const initBars = useTradingStore.getState().bars.toArray();
        if (initBars.length) series.setData(toLWData(initBars));

        // Subscribe to lastBarVersion WITHOUT re-rendering React.
        const unsub = useTradingStore.subscribe(
          (s) => s.lastBarVersion,
          () => {
            const arr = useTradingStore.getState().bars.toArray();
            series.setData(toLWData(arr));
            // Auto-scroll to rightmost edge unless user has panned (TODO Wave 3 userHasPanned)
            chart.timeScale().scrollToRealTime();
          },
        );

        return () => { unsub(); chart.remove(); };
      }, []);

      return (
        <div className="relative flex-1 min-w-[600px] bg-bg-base border-r border-border-subtle">
          <div ref={hostRef} className="absolute inset-0" />
          <ZoneOverlay chartRef={chartRef} />
        </div>
      );
    }

    function toLWData(bars: FootprintBar[]) {
      return bars.map(b => ({ ...b, time: b.ts as import('lightweight-charts').UTCTimestamp }));
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → WS server | JSON messages arrive over WebSocket; treated as untrusted until dispatched through the discriminated-union dispatcher |
| Canvas renderer → store | Direct store reads via `getState()` — store is trusted internal state |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-11 | Tampering | `useWebSocket` JSON.parse throws on malformed payload | mitigate | try/catch in onmessage drops malformed messages silently (does not crash the hook or stale the store) |
| T-11-12 | DoS | A flood of bars/signals from misbehaving backend could pin the React render cycle | mitigate | Canvas subscribes via `lastBarVersion` (no re-render); SignalFeed uses shadcn ScrollArea which virtualizes large lists cheaply; ring buffers cap memory at 500/200/50 entries |
| T-11-13 | Tampering | Malicious `levels` keys in bar payload (e.g. non-integer) | mitigate | FootprintRenderer uses `Number(tickKey)`; if `isNaN`, level is skipped (guard in the loop — explicit `if (Number.isFinite(price))`) |
| T-11-14 | Info Disclosure | No CSP | accept | Local dev per D-09 |
</threat_model>
