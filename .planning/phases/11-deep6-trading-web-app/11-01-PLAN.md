    **Step 1 — `deep6/api/ws_manager.py`:**

    ```python
    import asyncio, logging
    from typing import Any
    from fastapi import WebSocket
    from pydantic import BaseModel

    log = logging.getLogger(__name__)

    class WSManager:
        """Connection set + broadcast fan-out. Singleton in app.state.ws_manager.

        Per D-10: ONE WebSocket multiplexes all streams. Clients receive a mix
        of LiveBarMessage/LiveSignalMessage/LiveScoreMessage/LiveStatusMessage
        discriminated by the 'type' field.
        """

        def __init__(self) -> None:
            self.active: set[WebSocket] = set()
            self._lock = asyncio.Lock()

        async def connect(self, ws: WebSocket) -> None:
            await ws.accept()
            async with self._lock:
                self.active.add(ws)

        async def disconnect(self, ws: WebSocket) -> None:
            async with self._lock:
                self.active.discard(ws)

        async def broadcast(self, message: BaseModel | dict[str, Any]) -> None:
            """Fan-out one message to all active connections.

            - Accepts a Pydantic model or a dict.
            - Individual send failures do not block other recipients.
            - Failing sockets are discarded from active set.
            """
            payload = message.model_dump() if isinstance(message, BaseModel) else message
            # Snapshot under lock, then send without holding it.
            async with self._lock:
                targets = list(self.active)
            dead: list[WebSocket] = []
            for ws in targets:
                try:
                    await ws.send_json(payload)
                except Exception as e:   # noqa: BLE001 — per-socket failure must not abort broadcast
                    log.warning("ws send failed: %s", e)
                    dead.append(ws)
            if dead:
                async with self._lock:
                    for ws in dead:
                        self.active.discard(ws)
    ```

    import time
    from fastapi import APIRouter, WebSocket, WebSocketDisconnect, Request
    from deep6.api.schemas import (
        LiveBarMessage, LiveSignalMessage, LiveScoreMessage, LiveStatusMessage,
    )

    router = APIRouter(tags=["live"])

    @router.websocket("/ws/live")
    async def live_ws(websocket: WebSocket):
        manager = websocket.app.state.ws_manager
        await manager.connect(websocket)
        try:
            # Send initial connected status so client dot can turn green immediately.
            await websocket.send_json(LiveStatusMessage(
                connected=True, pnl=0.0, circuit_breaker_active=False,
                feed_stale=False, ts=time.time()
            ).model_dump())
            # Hold the connection. Client does not send messages; we just drain.
            while True:
                # receive_text blocks until client sends OR disconnects.
                await websocket.receive_text()
        except WebSocketDisconnect:
            pass
        finally:
            await manager.disconnect(websocket)

    # Debug/test helper — also usable by the Python trading engine to push
    # arbitrary live messages without importing WSManager directly.
    @router.post("/api/live/test-broadcast")

    ```python
    from fastapi import APIRouter, HTTPException, Request, Query
    from deep6.api.schemas import ReplayBarOut

    router = APIRouter(prefix="/api/replay", tags=["replay"])

    @router.get("/sessions")
    async def list_sessions(request: Request):
        store = request.app.state.event_store
        return await store.list_sessions()

    @router.get("/{session_id}")
    async def session_range(
        request: Request,
        session_id: str,
        start: int = Query(0, ge=0),
        end: int | None = Query(None, ge=0),
    ):
        """Prefetch a range of bars for replay scrubbing.
        Returns {session_id, total_bars, bars: [...]}.
        """
        store = request.app.state.event_store
        # Existence check via list_sessions
        sessions = await store.list_sessions()
        match = next((s for s in sessions if s["session_id"] == session_id), None)
        if match is None:
            raise HTTPException(status_code=404, detail="Session not found")
        bars = await store.fetch_bars_for_session(
            session_id, start_index=start, end_index=end,
        )
        return {
            "session_id": session_id,
            "total_bars": match["bar_count"],
            "bars": bars,
        }

    @router.get("/{session_id}/{bar_index}")
    async def replay_bar(request: Request, session_id: str, bar_index: int):
        """Returns one bar plus signals that fired at/before this bar's ts.

        Client uses this for step-through replay: advance bar_index, render.
        """
        store = request.app.state.event_store
        bar = await store.fetch_bar(session_id, bar_index)
        if bar is None:
            raise HTTPException(status_code=404, detail="Bar not found in session")
        # Signals up to and including bar.ts — slice via existing fetch_signal_events
        # then filter in-Python (signal_events has no session column; ts-based filter is correct).
        all_signals = await store.fetch_signal_events(limit=50000)
        signals_up_to = [s for s in all_signals if s["ts"] <= bar["ts"]]
        return {
            "session_id": session_id,
            "bar_index": bar_index,
            "bar": bar,
            "signals_up_to": signals_up_to,
        }
    ```

## STRIDE Threat Register

| T-11-01 | Tampering | `/api/live/test-broadcast` accepts arbitrary JSON and broadcasts to all WS clients | mitigate | Validate payload against `LiveMessage` Union discriminator before broadcast (parse with `TypeAdapter(LiveMessage).validate_python(payload)`); reject 422 on invalid shape so a malformed push cannot crash listeners |
| T-11-02 | Information Disclosure | `/api/replay/*` is unauthenticated | accept | Per D-09 localhost-only, trusted-network assumption; no PII in bar_history; doc this limit in plan summary so the deferred APP-07 auth phase knows to bind |
