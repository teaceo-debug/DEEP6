```ts
import {
  ICustomSeriesPaneView,
  ICustomSeriesPaneRenderer,
  CanvasRenderingTarget2D,
  PriceToCoordinateConverter,
  CustomSeriesPricePlotValues,
  DataChangedScope,
  UTCTimestamp,
} from "lightweight-charts"

export interface FootprintLevel {
  price: number
  bid: number
  ask: number
}

export interface FootprintData {
  time: UTCTimestamp
  open: number; high: number; low: number; close: number
  levels: FootprintLevel[]
  poc: number
  delta: number
  cvd: number
  tier: "TYPE_A" | "TYPE_B" | "TYPE_C" | "QUIET"
}

export interface FootprintOptions {
  cellHeight: number         // pixels per price tick, default 10
  bidColor: string           // default "#2563eb" (blue-600)
  askColor: string           // default "#dc2626" (red-600)
  pocColor: string           // default "#f59e0b" (amber-400)
  maxVolumeWidth: number     // max cell width as fraction of bar width, default 0.45
  showVolume: boolean        // show volume numbers in cells, default true
  fontSize: number           // default 9
}
```

FootprintSeriesRenderer (ICustomSeriesPaneRenderer):
- draw() method receives CanvasRenderingTarget2D and PriceToCoordinateConverter
- For each level in this._data.levels:
  - Convert level.price to y-coordinate via priceConverter
  - Draw bid rectangle on left half: width proportional to bid / maxLevelVolume * maxVolumeWidth * barWidth
    Fill with bidColor at 70% opacity; if level is POC → draw amber outline
  - Draw ask rectangle on right half: symmetric
  - If showVolume and cell is wide enough (>20px): draw volume number in white text
- Bar candle body: draw thin vertical line for OHLC (open/close range)
- TYPE_A bar: add amber glow border around the bar column

FootprintSeriesPlugin (ICustomSeriesPaneView<FootprintData, FootprintOptions>):
- Implements all required interface methods
- priceValueBuilder: returns [low, high, close] for scale computation
- isWhitespace: return !data.levels || data.levels.length === 0
- defaultOptions(): returns the default values above
- renderer(): returns the FootprintSeriesRenderer instance

Export:

   ```tsx
   "use client"
   import { useEffect, useRef } from "react"
   import { createChart, IChartApi } from "lightweight-charts"
   import { FootprintSeriesPlugin, FootprintData } from "@/lib/footprint-series"
   import { addZoneOverlays } from "@/lib/chart-overlays"
   import { useLiveStore } from "@/store/live"

   export default function FootprintChart() {
     const containerRef = useRef<HTMLDivElement>(null)
     const chartRef = useRef<IChartApi | null>(null)
     const signals = useLiveStore(s => s.signals)

     useEffect(() => {
       if (!containerRef.current) return

       // Create chart with dark theme
       const chart = createChart(containerRef.current, {
         layout: { background: { color: "#09090b" }, textColor: "#a1a1aa" },
         grid: { vertLines: { color: "#27272a" }, horzLines: { color: "#27272a" } },
         crosshair: { mode: 1 },
         rightPriceScale: { borderColor: "#3f3f46" },
         timeScale: { borderColor: "#3f3f46", timeVisible: true, secondsVisible: false },
       })
       chartRef.current = chart

       // Add custom footprint series
       const footprintSeries = chart.addCustomSeries(new FootprintSeriesPlugin(), {
         cellHeight: 10,
         bidColor: "#2563eb",
         askColor: "#dc2626",
         pocColor: "#f59e0b",
         maxVolumeWidth: 0.45,
         showVolume: true,
         fontSize: 9,
       })

       // Seed with synthetic demo bars (5 bars) so chart isn't blank on load
       // Each bar: time = now - (4-i)*60, open/high/low/close synthetic NQ price ~17000
       // levels: 10 price levels, each with random bid/ask volumes
       const seedBars: FootprintData[] = generateDemoBars(5)
       footprintSeries.setData(seedBars)

       // Add GEX overlay lines from latest signal's gex_regime
       // (static demo overlays — real data from signal events in production)
       addZoneOverlays(chart, {
         gexLevels: { gammaFlip: 17000, callWall: 17200, putWall: 16800 },
       })

       // ResizeObserver for responsive sizing
       const ro = new ResizeObserver(() => {
         if (containerRef.current) {
           chart.applyOptions({ width: containerRef.current.offsetWidth })
         }
       })
       ro.observe(containerRef.current)

       return () => { chart.remove(); ro.disconnect() }
     }, [])

     // TODO: subscribe to signals for live bar updates (Phase 11 wires real FootprintBar data)
     // For now: demo data only — real bar streaming comes with live Rithmic feed

     return <div ref={containerRef} className="w-full h-full min-h-[500px]" />
   }
   ```
