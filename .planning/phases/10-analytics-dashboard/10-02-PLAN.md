   ```ts
   export class WsClient {
     private ws: WebSocket | null = null
     private retryDelay = 1000      // start at 1s
     private maxDelay = 30000       // cap at 30s
     private shouldReconnect = true
     onMessage: (data: unknown) => void = () => {}
     onStatusChange: (s: WsStatus) => void = () => {}

     connect(token: string, url = "ws://localhost:8000/ws") {
       // new WebSocket(`${url}?token=${token}`)
       // onopen: retryDelay = 1000; onStatusChange("live")
       // onmessage: parse JSON; call this.onMessage(parsed)
       // onclose/onerror: onStatusChange("stale") then schedule reconnect
       // reconnect: exponential backoff: delay = min(delay * 2, maxDelay)
       // use setTimeout; do not reconnect if shouldReconnect=false
     }
     disconnect() { this.shouldReconnect = false; this.ws?.close() }
     send(data: unknown) { this.ws?.send(JSON.stringify(data)) }
   }

   export const wsClient = new WsClient()
   ```
