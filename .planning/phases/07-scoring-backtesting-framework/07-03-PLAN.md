Walk-forward fold structure:
```
Bars: [=====IN-SAMPLE=====][PURGE][===OOS===][PURGE][=====IN-SAMPLE=====][PURGE][===OOS===]
      fold 1 train         gap    fold 1 test gap    fold 2 train         gap    fold 2 test
```
- purge_bars: number of bars to exclude between in-sample and OOS (prevents leakage from lagged signals)
- Default: 5 folds, 20% OOS each, purge_bars=10

WFE formula:
```
WFE = mean(OOS P&L per fold) / mean(IS P&L per fold)
- WFE >= 0.70 → PASS → write best_params.json
- WFE < 0.70 → FAIL → sys.exit(1) with message "GATE FAILED: WFE={wfe:.2f} < 0.70"
```

    ```python
    Implements purged walk-forward efficiency (WFE) gate:
    - Split bars into N folds (train + OOS with purge gap)
    - Per fold: optimize thresholds on IS bars via Optuna (30 trials)
    ```

    ```python
    def split_folds(
        bars: list,
        n_folds: int = 5,
        oos_frac: float = 0.20,
        purge_bars: int = 10,
    ) -> list[tuple[list, list]]:
        """Split bars into (train, oos) fold pairs with purge gaps.

        Purge gap prevents leakage from signals with multi-bar lookback.
        Per D-10: purged splits prevent future leakage (TEST-05).

        Returns: list of (train_bars, oos_bars) tuples
        """
        n = len(bars)
        oos_size = int(n * oos_frac)
        folds = []

        for fold_idx in range(n_folds):
            # Walk forward: each fold's OOS is a non-overlapping window
            oos_start = n - (n_folds - fold_idx) * oos_size
            oos_end = oos_start + oos_size
            if oos_start <= 0:
                continue

            # Purge: exclude bars immediately before OOS start
            train_end = oos_start - purge_bars
            if train_end <= 10:
                continue

            train_bars = bars[:train_end]
            oos_bars = bars[oos_start:oos_end]

            if len(oos_bars) < 20:  # min meaningful OOS window
                continue

            folds.append((train_bars, oos_bars))

        return folds
    ```

    ```python
    def compute_wfe(is_pnls: list[float], oos_pnls: list[float]) -> float:
        """Walk-forward efficiency = mean(OOS) / mean(IS).

        Returns 0.0 if IS P&L is <= 0 (avoids division by near-zero).
        """
        mean_is = np.mean(is_pnls) if is_pnls else 0.0
        mean_oos = np.mean(oos_pnls) if oos_pnls else 0.0
        if mean_is <= 0:
            return 0.0
        return float(mean_oos / mean_is)


    def wfe_gate(wfe: float, threshold: float = 0.70) -> bool:
        """Return True if WFE meets threshold. False = GATE FAILED."""
        return wfe >= threshold
    ```

    ```python
    def run_fold(
        train_bars: list,
        oos_bars: list,
        n_trials: int = 30,
        fold_idx: int = 0,
    ) -> dict:
        """Optimize on IS, evaluate on OOS. Returns fold metrics."""
        print(f"  Fold {fold_idx+1}: IS={len(train_bars)} bars, OOS={len(oos_bars)} bars")

        # IS optimization
        study = optuna.create_study(
            direction="maximize",
            sampler=optuna.samplers.TPESampler(seed=42 + fold_idx),
        )
        study.optimize(make_objective(train_bars), n_trials=n_trials)

        best_params = study.best_trial.params
        is_pnl = study.best_value

        # Reconstruct config from best_params (handle missing fields with defaults)
        def get(key, default):
            return best_params.get(key, default)

        scorer_cfg = ScorerConfig(
            type_a_min=get("type_a_min", 80.0),
            type_b_min=get("type_b_min", 65.0),
            type_c_min=get("type_c_min", 50.0),
            confluence_threshold=int(get("confluence_threshold", 5)),
            zone_high_min=get("zone_high_min", 50.0),
            zone_high_bonus=get("zone_high_bonus", 8.0),
            zone_mid_bonus=get("zone_mid_bonus", 6.0),
        )
        abs_cfg = AbsorptionConfig(
            absorb_wick_min=get("absorb_wick_min", 30.0),
            absorb_delta_max=get("absorb_delta_max", 0.12),
            stop_vol_mult=get("stop_vol_mult", 2.0),
            evr_vol_mult=get("evr_vol_mult", 1.5),
        )
        imb_cfg = ImbalanceConfig(
            ratio_threshold=get("ratio_threshold", 3.0),
            stacked_t1=int(get("stacked_t1", 3)),
            stacked_t2=int(get("stacked_t2", 5)),
            stacked_t3=int(get("stacked_t3", 7)),
        )
        delta_cfg = DeltaConfig(
            tail_threshold=get("tail_threshold", 0.95),
            trap_delta_ratio=get("trap_delta_ratio", 0.3),
        )

        # OOS evaluation
        oos_results = run_backtest_with_configs(oos_bars, scorer_cfg, abs_cfg, imb_cfg, delta_cfg)
        tradeable_oos = [r for r in oos_results if r["tier"] in ("TYPE_A", "TYPE_B")]
        oos_pnl = sum(r["pnl_3bar"] for r in tradeable_oos) if tradeable_oos else 0.0

        print(f"    IS P&L={is_pnl:.2f}, OOS P&L={oos_pnl:.2f}, OOS signals={len(tradeable_oos)}")

        return {
            "fold": fold_idx + 1,
            "is_bars": len(train_bars),
            "oos_bars": len(oos_bars),
            "is_pnl": round(is_pnl, 2),
            "oos_pnl": round(oos_pnl, 2),
            "oos_signals": len(tradeable_oos),
            "best_params": best_params,
        }
    ```

    ```python
    def check_determinism(bars: list) -> bool:
        """TEST-07: Two runs on same bars + same config produce identical results."""
        cfg = ScorerConfig()
        abs_cfg = AbsorptionConfig()
        imb_cfg = ImbalanceConfig()
        delta_cfg = DeltaConfig()

        r1 = run_backtest_with_configs(bars[:100], cfg, abs_cfg, imb_cfg, delta_cfg)
        r2 = run_backtest_with_configs(bars[:100], cfg, abs_cfg, imb_cfg, delta_cfg)

        for i, (a, b) in enumerate(zip(r1, r2)):
            if abs(a["pnl_3bar"] - b["pnl_3bar"]) > 0.001:
                print(f"  DETERMINISM FAIL at bar {i}: run1={a['pnl_3bar']} run2={b['pnl_3bar']}")
                return False
            if a["tier"] != b["tier"]:
                print(f"  DETERMINISM FAIL at bar {i}: tier {a['tier']} vs {b['tier']}")
                return False
        print(f"  Determinism: PASS ({len(r1)} bars identical across 2 runs)")
        return True
    ```

    ```python
    def main():
        parser = argparse.ArgumentParser(description="DEEP6 Walk-Forward Validation")
        parser.add_argument("--start", required=True)
        parser.add_argument("--end", required=True)
        parser.add_argument("--folds", type=int, default=5)

        parser.add_argument("--output", default="best_params.json")
        parser.add_argument("--bar-seconds", type=int, default=60)
        parser.add_argument("--wfe-threshold", type=float, default=0.70)
        args = parser.parse_args()

        api_key = os.environ.get("DATABENTO_API_KEY", "")
        if not api_key:
            from dotenv import load_dotenv; load_dotenv()
            api_key = os.environ.get("DATABENTO_API_KEY", "")
        if not api_key:
            print("Set DATABENTO_API_KEY"); sys.exit(1)

        print(f"Fetching NQ trades {args.start} to {args.end}...")
        client = db.Historical(key=api_key)
        data = client.timeseries.get_range(
            dataset="GLBX.MDP3", schema="trades",
            stype_in="continuous", symbols=["NQ.c.0"],
            start=f"{args.start}T13:30:00", end=f"{args.end}T20:00:00",
        )
        bars = build_bars(data, bar_seconds=args.bar_seconds)
        print(f"Built {len(bars)} bars")

        # TEST-07: Determinism check first
        print("\nChecking signal determinism (TEST-07)...")
        det_ok = check_determinism(bars)
        if not det_ok:
            print("WARNING: Non-deterministic signals detected — investigate before continuing")

        # Split and run walk-forward
        print(f"\nRunning {args.folds}-fold walk-forward (purge=10 bars, {args.trials} trials/fold)...")
        folds = split_folds(bars, n_folds=args.folds, oos_frac=0.20, purge_bars=10)
        if not folds:
            print("ERROR: Not enough bars to create folds"); sys.exit(1)

        fold_results = []
        for i, (train, oos) in enumerate(folds):
            result = run_fold(train, oos, n_trials=args.trials, fold_idx=i)
            fold_results.append(result)

        # Compute WFE
        is_pnls = [f["is_pnl"] for f in fold_results]
        oos_pnls = [f["oos_pnl"] for f in fold_results]
        wfe = compute_wfe(is_pnls, oos_pnls)

        print(f"\n{'=' * 60}")
        print(f"WALK-FORWARD RESULTS: {len(fold_results)} folds")
        print(f"{'=' * 60}")
        print(f"{'Fold':<6} {'IS P&L':>10} {'OOS P&L':>10} {'OOS Sigs':>10}")
        for f in fold_results:
            print(f"{f['fold']:<6} {f['is_pnl']:>+10.2f} {f['oos_pnl']:>+10.2f} {f['oos_signals']:>10}")
        print(f"{'MEAN':<6} {np.mean(is_pnls):>+10.2f} {np.mean(oos_pnls):>+10.2f}")
        print(f"\nWFE = {wfe:.3f} (threshold: {args.wfe_threshold})")

        gate_pass = wfe_gate(wfe, args.wfe_threshold)
        if gate_pass:
            print(f"GATE PASSED: WFE={wfe:.3f} >= {args.wfe_threshold}")
            # Write best params from fold with highest OOS P&L
            best_fold = max(fold_results, key=lambda f: f["oos_pnl"])
            output = {
                "wfe": round(wfe, 4),
                "gate_threshold": args.wfe_threshold,
                "gate_status": "PASS",
                "best_fold": best_fold["fold"],
                "fold_count": len(fold_results),
                "params": best_fold["best_params"],
                "fold_breakdown": [
                    {"fold": f["fold"], "is_pnl": f["is_pnl"], "oos_pnl": f["oos_pnl"]}
                    for f in fold_results
                ],
            }
            with open(args.output, "w") as fp:
                json.dump(output, fp, indent=2)
            print(f"Wrote {args.output}")
        else:
            print(f"GATE FAILED: WFE={wfe:.3f} < {args.wfe_threshold}")
            print("No best_params.json written. Investigate before rerunning.")
            sys.exit(1)

    if __name__ == "__main__":
        main()
    ```

    ```python
    """Tests for walk_forward.py — fold splitting, purge logic, WFE gate.

    No Databento required — uses lightweight bar stubs.
    """
    import pytest
    import sys, os
    sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

    from scripts.walk_forward import split_folds, compute_wfe, wfe_gate


    def fake_bars(n: int) -> list:
        """Return n minimal bar-like objects sufficient for split_folds."""
        # split_folds only needs len(bars) and slicing — any list works
        return list(range(n))  # integers suffice for fold splitting tests


    def test_split_folds_count():
        bars = fake_bars(100)
        folds = split_folds(bars, n_folds=3, oos_frac=0.20, purge_bars=5)
        assert len(folds) == 3, f"Expected 3 folds, got {len(folds)}"


    def test_purge_gap():
        """IS end index must be at least purge_bars before OOS start index."""
        bars = fake_bars(200)
        purge = 10
        folds = split_folds(bars, n_folds=3, oos_frac=0.20, purge_bars=purge)
        for train, oos in folds:
            # bars are integers — train[-1] is the last IS bar index, oos[0] is first OOS
            # The gap: oos[0] - train[-1] must be > purge_bars
            # Since bars are range(200), train[-1] + 1 is len(train), oos[0] is the index value
            assert train[-1] + purge < oos[0], (
                f"Purge gap violated: IS ends at {train[-1]}, OOS starts at {oos[0]}, purge={purge}"
            )


    def test_oos_non_overlapping():
        bars = fake_bars(200)
        folds = split_folds(bars, n_folds=3, oos_frac=0.20, purge_bars=5)
        oos_ranges = [(oos[0], oos[-1]) for _, oos in folds]
        for i in range(len(oos_ranges) - 1):
            end_i = oos_ranges[i][1]
            start_j = oos_ranges[i + 1][0]
            assert end_i < start_j, f"OOS overlap: fold {i} ends {end_i}, fold {i+1} starts {start_j}"


    def test_compute_wfe_normal():
        wfe = compute_wfe([100.0, 80.0, 90.0], [75.0, 60.0, 70.0])
        expected = (75 + 60 + 70) / 3 / ((100 + 80 + 90) / 3)
        assert abs(wfe - expected) < 0.001


    def test_compute_wfe_zero_is():
        wfe = compute_wfe([-10.0, -20.0], [5.0, 5.0])
        assert wfe == 0.0, "WFE should be 0 when IS mean <= 0"


    def test_wfe_gate_pass():
    ```
