**Create `deep6/state/dom.py`** — DOMState pre-allocated arrays (D-13):
```python
"""DOMState: pre-allocated array.array for zero-allocation DOM callbacks."""
import array
import time
from dataclasses import dataclass, field

LEVELS = 40  # 40 bid + 40 ask levels pre-allocated

@dataclass
class DOMState:
    bid_prices: array.array = field(default_factory=lambda: array.array('d', [0.0] * LEVELS))
    bid_sizes:  array.array = field(default_factory=lambda: array.array('d', [0.0] * LEVELS))
    ask_prices: array.array = field(default_factory=lambda: array.array('d', [0.0] * LEVELS))
    ask_sizes:  array.array = field(default_factory=lambda: array.array('d', [0.0] * LEVELS))
    last_update: float = 0.0

    def update(
        self,
        bid_prices: list,
        bid_sizes: list,
        ask_prices: list,
        ask_sizes: list,
        ts: float | None = None,
    ) -> None:
        """In-place update — zero allocation per call. Called from DOM callback."""
        n_bid = min(len(bid_prices), LEVELS)
        n_ask = min(len(ask_prices), LEVELS)
        for i in range(n_bid):
            self.bid_prices[i] = bid_prices[i]
            self.bid_sizes[i]  = bid_sizes[i]
        for i in range(n_ask):
            self.ask_prices[i] = ask_prices[i]
            self.ask_sizes[i]  = ask_sizes[i]
        self.last_update = ts if ts is not None else time.monotonic()

    def snapshot(self) -> tuple:
        """Copy for engine use — called once per bar close, NOT per callback."""
        return (
            list(self.bid_prices), list(self.bid_sizes),
            list(self.ask_prices), list(self.ask_sizes),
        )
```

"""Rithmic connection factory and aggressor verification gate."""
import asyncio
import structlog
from async_rithmic import RithmicClient, DataType, ReconnectionSettings
from deep6.config import Config

log = structlog.get_logger()

# Aggressor gate state — module-level (single event loop, no threading needed)
_aggressor_sample_count: int = 0
_aggressor_unknown_count: int = 0
_aggressor_verified: bool = False


async def connect_rithmic(config: Config) -> RithmicClient:
    """Connect to Rithmic with Issue #49 workaround (sequential plant delay).
    
    Per D-18: sequential plant connection on reconnect (ForcedLogout bug workaround).
    Per D-19: all connect/disconnect events logged with timestamps.
    """
    client = RithmicClient(
        user=config.rithmic_user,
        password=config.rithmic_password,
        system_name=config.rithmic_system_name,
        app_name="DEEP6",
        app_version="2.0.0",
        uri=config.rithmic_uri,
        reconnection_settings=ReconnectionSettings(
            max_retries=10,
            base_delay=1.0,
            max_delay=60.0,
            backoff_factor=2.0,
            jitter=True,
        ),
    )
    await client.connect()
    # Issue #49 workaround: 500ms delay after connect before subscribing (D-18)
    await asyncio.sleep(0.5)
    log.info("rithmic.connected", uri=config.rithmic_uri, system=config.rithmic_system_name)
    return client


def register_callbacks(client: RithmicClient, state: "SharedState") -> None:
    """Attach all event callbacks to the connected client."""
    from deep6.data.dom_feed import make_dom_callback
    from deep6.data.tick_feed import make_tick_callback
    client.on_order_book += make_dom_callback(state)
    client.on_tick += make_tick_callback(state)
    client.on_connected += lambda: log.info("rithmic.on_connected")
    client.on_disconnected += lambda: _on_disconnected(state)


def _on_disconnected(state: "SharedState") -> None:
    """Activate FROZEN state immediately on disconnect (D-17, D-19)."""
    import time
    ts = time.time()
    log.warning("rithmic.disconnected", ts=ts)
    state.freeze_guard.on_disconnect(ts)


async def aggressor_verification_gate(config: Config) -> bool:
    """Sample N ticks and confirm TransactionType.BUY/SELL is present.
    
    Returns True when gate passes (verified non-UNSPECIFIED aggressor observed).
    Returns False and logs error if >aggressor_max_unknown_pct are UNSPECIFIED.
    
    CRITICAL per D-03: No FootprintBar accumulator code runs until this gate passes.
    TransactionType: TRANSACTIONTYPE_UNSPECIFIED=0, BUY=1, SELL=2
    """
    global _aggressor_sample_count, _aggressor_unknown_count, _aggressor_verified
    if _aggressor_verified:
        return True
    if _aggressor_sample_count < config.aggressor_sample_size:
        return False  # still sampling
    unknown_pct = _aggressor_unknown_count / _aggressor_sample_count
    if unknown_pct > config.aggressor_max_unknown_pct:
        log.error(
            "aggressor.verification_failed",
            sample_count=_aggressor_sample_count,
            unknown_count=_aggressor_unknown_count,
            unknown_pct=f"{unknown_pct:.1%}",
            action="ESCALATE — footprint accumulation disabled. Check Rithmic CME aggressor feed.",
        )
        return False
    _aggressor_verified = True
    log.info(
        "aggressor.verified",
        sample_count=_aggressor_sample_count,
        unknown_count=_aggressor_unknown_count,
        unknown_pct=f"{unknown_pct:.1%}",
    )
    return True


def record_aggressor_sample(aggressor: int) -> None:
    """Called from tick_feed for each LAST_TRADE tick during gate sampling."""
    global _aggressor_sample_count, _aggressor_unknown_count
    _aggressor_sample_count += 1
    if aggressor == 0:  # TRANSACTIONTYPE_UNSPECIFIED
        _aggressor_unknown_count += 1
```

"""DOM callback factory — O(1) per callback, zero allocation."""
import structlog
log = structlog.get_logger()

def make_dom_callback(state: "SharedState"):
    """Returns the async on_order_book callback bound to shared state."""
    async def on_order_book(update) -> None:
        # Only process SOLO and END updates (book is evaluable)
        update_type = getattr(update, "update_type", None)
        if update_type not in ("SOLO", "END"):
            return
        if state.freeze_guard.is_frozen:
            return  # D-17: no processing during FROZEN state
        # Extract bid/ask arrays from update object
        bid_prices = [l.price for l in update.bids] if update.bids else []
        bid_sizes  = [l.size  for l in update.bids] if update.bids else []
        ask_prices = [l.price for l in update.asks] if update.asks else []
        ask_sizes  = [l.size  for l in update.asks] if update.asks else []
        state.dom.update(bid_prices, bid_sizes, ask_prices, ask_sizes)
    return on_order_book
```

**Create `deep6/data/tick_feed.py`** — tick callback factory (DATA-02, D-03):
```python
"""Tick callback factory — LAST_TRADE drives footprint accumulator."""
import structlog
from async_rithmic import DataType
from deep6.data.rithmic import record_aggressor_sample, aggressor_verification_gate

log = structlog.get_logger()

def make_tick_callback(state: "SharedState"):
    """Returns the async on_tick callback bound to shared state."""
    async def on_tick(tick) -> None:
        if state.freeze_guard.is_frozen:
            return  # D-17: no processing during FROZEN state
        if not hasattr(tick, "data_type"):
            return
        if tick.data_type == DataType.LAST_TRADE:
            lt = tick.last_trade
            aggressor = getattr(lt, "aggressor", 0)
            # Record for verification gate (D-03)
            record_aggressor_sample(aggressor)
            # Only accumulate to footprint once gate is verified (D-03)
            gate_passed = await aggressor_verification_gate(state.config)
            if gate_passed and aggressor != 0:
                # Forward to bar builders (attached to state in Plan 02)
                for builder in state.bar_builders:
                    builder.on_trade(lt.price, lt.size, aggressor)
    return on_tick
```
