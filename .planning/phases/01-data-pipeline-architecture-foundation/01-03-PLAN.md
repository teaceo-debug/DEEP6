"""SessionPersistence: async SQLite key-value store for session state.

Per D-15: aiosqlite for async SQLite access.
Per D-07: session_id is the date string "YYYYMMDD"; keys are session state fields.
Schema: session_state(session_id TEXT, key TEXT, value TEXT, updated_at REAL)
"""
import time
import aiosqlite

SCHEMA = """
CREATE TABLE IF NOT EXISTS session_state (
    session_id TEXT NOT NULL,
    key        TEXT NOT NULL,
    value      TEXT NOT NULL,
    updated_at REAL NOT NULL,
    PRIMARY KEY (session_id, key)
);
"""

SESSION_KEYS = [
    "cvd", "vwap_numerator", "vwap_denominator",
    "ib_high", "ib_low", "ib_complete",
    "opening_range_high", "opening_range_low", "day_type",
]


class SessionPersistence:
    """Async key-value store backed by SQLite. One row per (session_id, key).
    
    Usage:
        p = SessionPersistence("./deep6_session.db")
        await p.initialize()
        await p.write("20260411", "cvd", "42")
        data = await p.read_all("20260411")
        ctx = SessionContext.from_dict(data)
    """

    def __init__(self, db_path: str) -> None:
        self.db_path = db_path

    async def initialize(self) -> None:
        """Create table if not exists. Call once at startup."""
        async with aiosqlite.connect(self.db_path) as db:
            await db.execute(SCHEMA)
            await db.commit()

    async def write(self, session_id: str, key: str, value: str) -> None:
        """Insert or replace a session state entry."""
        async with aiosqlite.connect(self.db_path) as db:
            await db.execute(
                "INSERT OR REPLACE INTO session_state "
                "(session_id, key, value, updated_at) VALUES (?, ?, ?, ?)",
                (session_id, key, value, time.time()),
            )
            await db.commit()

    async def write_many(self, session_id: str, data: dict) -> None:
        """Write all key-value pairs from a dict atomically."""
        now = time.time()
        async with aiosqlite.connect(self.db_path) as db:
            await db.executemany(
                "INSERT OR REPLACE INTO session_state "
                "(session_id, key, value, updated_at) VALUES (?, ?, ?, ?)",
                [(session_id, k, v, now) for k, v in data.items()],
            )
            await db.commit()

    async def read_all(self, session_id: str) -> dict:
        """Return all key-value pairs for the given session_id."""
        async with aiosqlite.connect(self.db_path) as db:
            async with db.execute(
                "SELECT key, value FROM session_state WHERE session_id = ?",
                (session_id,),
            ) as cursor:
                return {row[0]: row[1] async for row in cursor}

    async def persist_session_context(self, session_id: str, ctx) -> None:
        """Convenience: persist a SessionContext to SQLite."""
        await self.write_many(session_id, ctx.to_dict())

    async def restore_session_context(self, session_id: str):
        """Convenience: restore a SessionContext from SQLite.
        
        Returns None if no state exists for this session_id.
        """
        from deep6.state.session import SessionContext
        data = await self.read_all(session_id)
        if not data:
            return None
        return SessionContext.from_dict(data)

**Create `deep6/state/connection.py`** — FreezeGuard + ConnectionState + SessionManager:

```python
"""Connection state management: FROZEN guard and RTH/GC session management.

FreezeGuard (D-17, D-18, D-19):
  - Enters FROZEN on disconnect — all callback processing halted
  - Sequential plant reconnection with 500ms delay (Issue #49 workaround)
  - Exits FROZEN only after position reconciliation (stub for Phase 8)

SessionManager (D-16, D-07):
  - Disables GC at RTH open (9:30 ET), enables at RTH close (16:00 ET)
  - Resets SessionContext at RTH open
  - Persists SessionContext to SQLite at RTH close
"""
import asyncio
import gc
import time
from datetime import datetime
from zoneinfo import ZoneInfo

import structlog

log = structlog.get_logger()
EASTERN = ZoneInfo("America/New_York")


class ConnectionState:
    CONNECTED    = "CONNECTED"
    FROZEN       = "FROZEN"
    RECONNECTING = "RECONNECTING"


class FreezeGuard:
    """Guards all state mutation during disconnect/reconnect cycle.
    
    Every callback checks is_frozen before touching shared state.
    FROZEN → RECONNECTING → CONNECTED transition logged with timestamps (D-19).
    """

    def __init__(self) -> None:
        self._state: str = ConnectionState.CONNECTED

    @property
    def is_frozen(self) -> bool:
        return self._state != ConnectionState.CONNECTED

    def on_disconnect(self, ts: float | None = None) -> None:
        """Immediately enter FROZEN on any disconnect (D-17)."""
        ts = ts or time.time()
        self._state = ConnectionState.FROZEN
        log.warning(
            "connection.disconnected",
            ts=ts,
            state=self._state,
            action="FROZEN — no bar processing until reconnect + reconciliation",
        )

    async def on_reconnect(self, client, config) -> None:
        """Reconnect with sequential plant delay (D-18), then unfreeze (D-19).
        
        Position reconciliation stub: for Phase 8, this will query Rithmic
        position state before setting CONNECTED. Currently a placeholder.
        """
        self._state = ConnectionState.RECONNECTING
        log.info("connection.reconnecting", state=self._state)

        # Issue #49 workaround: sequential plant connection with 500ms delay (D-18)
        await client.connect()
        await asyncio.sleep(0.5)

        # TODO Phase 8: await sync_position_state(client) before unfreezing
        self._state = ConnectionState.CONNECTED
        log.info(
            "connection.restored",
            state=self._state,
            ts=time.time(),
        )

    def get_state(self) -> str:
        return self._state


class SessionManager:
    """Manages RTH session lifecycle: GC control, state reset, persistence.
    
    Per D-16: gc.disable() at 9:30 ET, gc.collect() at session open and close.
    Per D-07: SessionContext.reset() at 9:30 ET each day.
    Polls every 1 second — negligible CPU cost.
    """

    def __init__(self, state: "SharedState") -> None:
        self.state = state
        self._session_active: bool = False

    def _is_rth(self) -> bool:
        now_et = datetime.now(EASTERN)
        return (
            (now_et.hour == 9 and now_et.minute >= 30)
            or (10 <= now_et.hour < 16)
        )

    def _session_id(self) -> str:
        """Session ID = UTC date string YYYYMMDD."""
        from datetime import timezone
        return datetime.now(timezone.utc).strftime("%Y%m%d")

    async def run(self) -> None:
        """Background task: poll RTH boundaries every 1 second."""
        while True:
            is_rth = self._is_rth()
            if is_rth and not self._session_active:
                await self._on_session_open()
            elif not is_rth and self._session_active:
                await self._on_session_close()
            await asyncio.sleep(1.0)

    async def _on_session_open(self) -> None:
        """RTH open: reset state, disable GC (D-16, D-07)."""
        self._session_active = True
        gc.collect()   # clean sweep before disabling GC
        gc.disable()   # D-16: no GC during RTH
        self.state.session.reset()
        # Attempt to restore prior session state (mid-session restart, D-07)
        sid = self._session_id()
        restored = await self.state.persistence.restore_session_context(sid)
        if restored is not None:
            self.state.session.cvd = restored.cvd
            self.state.session.vwap_numerator = restored.vwap_numerator
            self.state.session.vwap_denominator = restored.vwap_denominator
            self.state.session.ib_high = restored.ib_high
            self.state.session.ib_low = restored.ib_low
            self.state.session.ib_complete = restored.ib_complete
            log.info("session.restored_from_db", session_id=sid, cvd=restored.cvd)
        else:
            log.info("session.fresh_start", session_id=sid)

    async def _on_session_close(self) -> None:
        """RTH close: persist state, re-enable GC (D-16, D-07)."""
        self._session_active = False
        sid = self._session_id()
        await self.state.persistence.persist_session_context(sid, self.state.session)
        log.info("session.persisted", session_id=sid, cvd=self.state.session.cvd)
        gc.enable()
        gc.collect()   # D-16: manual collect at session end
```

```python
"""SharedState: single container object passed to all callbacks and coroutines.

No locks needed — single asyncio event loop thread owns all state.
All callback functions receive 'state' as their only context argument.
Assembled once in __main__.py; never reconstructed during a session.
"""
from __future__ import annotations
import asyncio
from dataclasses import dataclass, field
from collections import deque
from typing import Callable, Awaitable

from deep6.config import Config
from deep6.state.dom import DOMState
from deep6.state.session import SessionContext
from deep6.state.connection import FreezeGuard, SessionManager
from deep6.state.persistence import SessionPersistence
from deep6.signals.atr import ATRTracker


@dataclass
class SharedState:
    """All shared state for DEEP6. Assembled once at startup.
    
    bar_builders: list populated by __main__.py after BarBuilder creation.
    atr_trackers: dict keyed by label ("1m", "5m").
    on_bar_close: async callback invoked by BarBuilder at each bar close.
                  Signature: async def on_bar_close(label: str, bar: FootprintBar)
                  In Phase 1 this is a no-op. Signal engines attach in Phase 2+.
    """
    config: Config
    dom: DOMState = field(default_factory=DOMState)
    session: SessionContext = field(default_factory=SessionContext)
    freeze_guard: FreezeGuard = field(default_factory=FreezeGuard)
    persistence: SessionPersistence = field(default=None)  # set in build()

    # Bar builders (1m and 5m) — populated by __main__.py
    bar_builders: list = field(default_factory=list)

    # ATR trackers keyed by label
    atr_trackers: dict = field(default_factory=lambda: {
        "1m": ATRTracker(period=20),
        "5m": ATRTracker(period=20),
    })

    # on_bar_close callback — replaced by signal pipeline in Phase 2+
    # Default: log and return (no-op)
    _on_bar_close_fn: Callable | None = field(default=None, repr=False)

    async def on_bar_close(self, label: str, bar) -> None:
        """Dispatched by BarBuilder at each bar close.
        
        In Phase 1: logs bar close and does nothing else.
        In Phase 2+: engines attach here via __main__.py setup.
        """
        import structlog
        log = structlog.get_logger()
        log.debug(
            "bar.closed",
            label=label,
            timestamp=bar.timestamp,
            close=bar.close,
            bar_delta=bar.bar_delta,
            cvd=bar.cvd,
            poc=bar.poc_price,
            total_vol=bar.total_vol,
        )
        if self._on_bar_close_fn is not None:
            await self._on_bar_close_fn(label, bar)

    @classmethod
    def build(cls, config: Config) -> "SharedState":
        """Factory: build SharedState with all sub-components wired up."""
        persistence = SessionPersistence(config.db_path)
        state = cls(config=config, persistence=persistence)
        return state

    def session_manager(self) -> "SessionManager":
        """Return a SessionManager bound to this state (for use in asyncio.gather)."""
        return SessionManager(self)
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| async-rithmic disconnect event → FreezeGuard | on_disconnected callback must immediately set FROZEN before any other callback runs |
| SQLite disk → session state | DB file contains session state but no credentials; low sensitivity |
| gc.disable() → event loop | GC disabled during RTH; if a coroutine panics and GC stays disabled, memory leaks permanently |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
