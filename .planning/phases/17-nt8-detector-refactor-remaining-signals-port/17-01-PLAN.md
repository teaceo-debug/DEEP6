namespace NinjaTrader.NinjaScript.AddOns.DEEP6.Registry {
    public interface ISignalDetector {
        string Name { get; }
        void Reset();                                             // session-boundary reset
        SignalResult[] OnBar(FootprintBar bar, SessionContext s); // bar-close evaluation
    }

    public sealed class SignalResult {
        public string SignalId;    // "IMB-01", "DELT-10", etc.
        public int Direction;      // +1 long, -1 short, 0 neutral
        public double Strength;    // 0.0-1.0
        public ulong FlagBit;      // single-bit ulong mask
        public string Detail;      // diagnostic
    }
}
