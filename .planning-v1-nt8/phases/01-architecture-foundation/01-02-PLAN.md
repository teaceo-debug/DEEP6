    // Welford online algorithm: add one sample and update mean/variance
    // Returns: (mean, sample_std_dev)
    private static (double mean, double std) WelfordAdd(double x, ref double m, ref double m2, ref int n)
    {
        n++;
        double delta = x - m;
        m += delta / n;
        double delta2 = x - m;
        m2 += delta * delta2;
        double std = n > 1 ? Math.Sqrt(m2 / (n - 1)) : 0.0;
        return (m, std);
    }
    ```

    // Zero-allocation mean/std using foreach over Queue<double> — no ToArray, no LINQ
    private static void QueueStats(Queue<double> q, out double mean, out double std)
    {
        int n = q.Count;
        if (n == 0) { mean = 0; std = 0; return; }
        double sum = 0;
        foreach (double v in q) sum += v;
        mean = sum / n;
        if (n < 2) { std = 0; return; }
        double ssq = 0;
        foreach (double v in q) ssq += (v - mean) * (v - mean);
        std = Math.Sqrt(ssq / (n - 1));
    }
    ```
